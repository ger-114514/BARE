        hidden_effect = {
            meta_effect = {
                text = {
                    deactivate_shine_on_focus = [FOCUS]

                        focus = [FOCUS]
                        use_side_message = yes
                        originator_name = "[ORIGINATOR]"

                    if = {
                        limit = { 
                            NOT = { check_variable = { TRI_hoshubu_focus_line^1 = 0 } }
                        }
                        activate_shine_on_focus = [NEW_FOCUS]
                        country_event = {
                            id = TRI_autofocus.1
                            days = var:TRI_auto_focus_days
                        }
                    }
                    remove_from_array = {
                        array = TRI_hoshubu_focus_line
                        index = 0
                    }
                }
                FOCUS = "[?TRI_hoshubu_focus_line^0.GetTokenKey]"
                ORIGINATOR = "[?TRI_auto_focus_originator.GetTokenKey]"
                NEW_FOCUS = "[?TRI_hoshubu_focus_line^1.GetTokenKey]"
            }
        }
        
    }
}
