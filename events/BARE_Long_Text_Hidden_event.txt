#阿里乌斯夫人胜利线事件
country_event = {
    id = bare_long_text_hidden_event.1
    title = bare_long_text_hidden_event.1.t
    desc = bare_long_text_hidden_event.1.d

	is_triggered_only = yes
	hidden = yes

	immediate = {
		clr_country_flag = BARE_long_event_show
		set_variable = {
			BARE_long_event_id = 1
		}
		resize_array = {
			array = BARE_long_event_select_array
			size = 0
		}
		add_to_array = { #选项号，最多2个
			BARE_long_event_select_array = 1
		}
		#为了防止玩家忘关事件走事件，先关事件然后延迟3小时开
		country_event = {
			id = bare_long_text_hidden_event.114514
			hours = 3
		}
		#这个就是夫人事件的效果，与框架无关
		country_event = {
			id = bare_long_text_hidden_event.2
			days = 3
		}

	}
    option = {
		name = bare_hidden_event.1.a
		ai_chance = {
			base = 100
		}
    }
}
