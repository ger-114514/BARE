BA_TRI_character_can_be_purged = {
	NOT = {
		#已经被抓
		has_character_flag = BA_TRI_character_arrested
		#重要学生
		is_character = TRI_seia
		is_character = TRI_nagisa
		is_character = TRI_mika
		is_character = TRI_tsurugi
		is_character = TRI_hasumi
		is_character = TRI_sakurako
		is_character = TRI_mine
		#补习部
		is_character = TRI_azusa
		is_character = TRI_hanako
		is_character = TRI_koharu
		is_character = TRI_hifumi
	} 
}
BA_TRI_can_purge_administration = {
	OR = {
		any_character = {
			OR = {
				is_character_slot = political_advisor
				is_character_slot = theorist
			}
			BA_TRI_character_can_be_purged = yes
		}
		NOT = { has_idea = BA_TRI_inspected_administration }
		TRI_nagisa = { NOT = { has_trait = BA_TRI_nagisa_paranoid } }
	}
}
BA_TRI_can_purge_army = {
	OR = {
		any_army_leader = {
			BA_TRI_character_can_be_purged = yes
		}
		any_character = {
			OR = {
				is_character_slot = army_chief
				is_character_slot = high_command
				is_character_slot = air_chief
			}
			BA_TRI_character_can_be_purged = yes
		}
		NOT = { has_idea = BA_TRI_inspected_army }
		NOT = { has_idea = BA_TRI_training_activities_curtailed }
	}
}
BA_TRI_can_purge_airforce = {
	OR = {
		NOT = { has_idea = BA_TRI_inspected_airforce }
		NOT = { has_idea = BA_TRI_ground_based_training }
	}
}
