on_actions = {
	on_puppet = {
		effect = {
			ROOT = {
				every_controlled_state = {
					ROOT = {
						give_resource_rights = {
							receiver = overlord
							state = PREV
						}
					}
				}
			}
		}
	}
	on_subject_free = {
		effect = {
			ROOT = {
				every_controlled_state = {
					ROOT = { remove_resource_rights = PREV }
				}
			}
		}
	}
}
