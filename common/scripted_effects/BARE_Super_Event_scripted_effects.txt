	hidden_effect = {
		set_variable_to_random = {
			var = BARE_SuperEvent_2_random_hour
			min = 8
			max = 23
			integer = yes
		}
		set_variable_to_random = {
			var = BARE_SuperEvent_2_random_minite
			min = 10
			max = 59
			integer = yes
		}
		log = "BARE_SuperEvent_2_temp_var [?BARE_SuperEvent_2_temp_var]"
		set_variable = { BARE_SuperEvent_2_var = BARE_SuperEvent_2_temp_var }
	}
}
