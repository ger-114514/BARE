	ARI_suppression_window = {
	    context_type = decision_category
		window_name = "ARI_suppression_window"

		visible = {
			has_country_flag = ARI_suppression_available
		}
		
		dynamic_lists = {
			ARI_suppression_states = {
				array = ARI_states
				value = state_id
				index = i
				entry_container = ARI_state_button_container
			}
		}
		triggers = {
			ARI_state_button_visible = {
				var:state_id = { NOT = { has_state_flag = ARI_state_absolutely_controlled } }
			}
		}
		effects = {
			ARI_state_button_click = { 
				set_variable = { ARI_map_selected_state = var:state_id }
			}
			ARI_suppression_map_mode_control_button_click = {
				set_variable = { ARI_suppression_map_mode_control = 2 }
				set_variable = { ARI_suppression_map_mode_guard = 1 }
				set_variable = { ARI_suppression_map_mode = token:ALR_resistance_state }
				every_owned_state = {
					set_variable = { ARI_state_button_frame = ARI_state_control_rank }
				}
			}
			ARI_suppression_map_mode_guard_button_click = {
				set_variable = { ARI_suppression_map_mode_control = 1 }
				set_variable = { ARI_suppression_map_mode_guard = 2 }
				set_variable = { ARI_suppression_map_mode = token:ARI_suppression_state_guard }
				every_owned_state = {
					set_variable = { ARI_state_button_frame = ARI_state_guard_rank }
				}
			}
		}
		properties = {
			ARI_state_button = {
				image = "[?ARI_suppression_map_mode.GetTokenKey]_[?state_id.GetID]"
				frame = var:state_id:ARI_state_button_frame
				x = ARI_states_x^i
				y = ARI_states_y^i
			}
			ARI_suppression_control_bar = {
				frame = var:ARI_map_selected_state:ARI_state_control
			}
			ARI_suppression_map_mode_control_button = {
				frame = var:ARI_suppression_map_mode_control
			}
			ARI_suppression_guard_bar = {
				frame = var:ARI_map_selected_state:ARI_state_guard
			}
			ARI_suppression_guard_rank_bar = {
				frame = var:ARI_map_selected_state:ARI_state_guard_rank
			}
			ARI_suppression_map_mode_guard_button = {
				frame = var:ARI_suppression_map_mode_guard
			}
		}
		
	}
